t = int(input()) for _ in range(t):    n = int(input())     arr = list(map(int, input().split()))         res = []     l = 0     r = n - 1     for i in range(n):         if i % 2 == 0:            res.append(arr[l])            l += 1         else:            res.append(arr[r])            r -= 1     print(*res)
