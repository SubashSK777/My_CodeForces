t = int(input()) for _ in range(t):     n = int(input())     arr = list(map(int, input().split()))     eve = []     odd = []     res = []     for i in arr:        if i % 2 == 0:            eve.append(i)         else:            odd.append(i)     eve.sort()     odd.sort()     i, j, k = 0,0,0     while i < n:        if arr[i] % 2 == 0:            res.append(eve[j])            i += 1            j += 1         else:            res.append(odd[k])            k += 1            i += 1     print("YES") if res == sorted(arr) else print("NO")
